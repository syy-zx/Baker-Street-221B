```C++
using ll = unsigned long long;
ll quickMulti(ll a, ll b) 
{
    ll ans = 0;
    for (; b; b >>= 1) 
    {
        if (b & 1) 
        {
            ans += a;
        }
        a <<= 1;
    }
    return ans;
}
```
